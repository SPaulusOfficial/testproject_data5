<p>-- Mapping Entries (encrypted original values)<br>CREATE TABLE mapping_entries (<br>    id UUID PRIMARY KEY,<br>    transformation_id UUID REFERENCES transformations(id),<br>    placeholder VARCHAR(100) NOT NULL,<br>    encrypted_value TEXT NOT NULL,<br>    category VARCHAR(100),<br>    confidence DECIMAL(3,2)<br>);</p>
